[docs]class FS(tornado.web.RequestHandler):
    """FS class reads through files of filesystem located under root/
    and helps to display the file contents during search results or when user requests to view file

    :func:`get` handles the root/ path sent.

    - **parameters**, **types**, **return** and **return types**::
        :parameter: arg, root/ path
	:type: string
        :return: file contents based on request
    
    :func:`_async_callback` gets file meta information

    - **parameters**, **types**, **return** and **return types**::
        :parameter: items, list of files in tree
        :type: list
        :return: gathers file meta information

    """
    @tornado.web.asynchronous
    def get(self, arg):
	self.root = options.__ROOT__
	if not arg:
		arg = self.root
		self.path = self.root
	else:
		self.path = self.root + str(arg)
	if os.path.isfile(self.path):
		f = open(self.path, 'r')
		self.write(tornado.escape.xhtml_escape(f.read()))
		f.close()
		self.finish()
	else:
		items = os.listdir(self.path)
        	self._async_callback(items)
    
    def _async_callback(self, items):
	fs = {}
	count = 0; size=0
	for f in items:
		fs[count] = []
		mode = os.stat(self.path + f).st_mode
		if S_ISDIR(mode):
        		mode = 'dir'
			url = "http://localhost:8888/" + self.path + f + "/"
		else:
			mode = 'file'
			url = "http://localhost:8888/" + self.path + f
		size = os.stat(self.path + f)[6]
		mtime = time.ctime(os.path.getmtime(self.path + f))
		fs[count]=[f, mode, size, mtime, url]
		count += 1
        self.render('fsMgr.html',dict=fs)


class Meta(tornado.web.RequestHandler):
    """Meta class provides meta data information of filesystem under root/

    :func:`get` handles the root/ path sent.

    - **parameters**, **types**, **return** and **return types**::
        
        :return: file system meta data information
        
    """

    def get(self):
	x,y = MetaData(options.__ROOT__)
	root = options.__ROOT__
	metadata = {
		"files":x,
		"directories":y,
		"root":root
	}
	self.write(tornado.escape.json_encode(metadata))
