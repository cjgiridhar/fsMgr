    """This class provides simple API wrapper over Whoosh search engine.

    :func:`__init__` - creates a directory where index is stored.

    - **parameters**, **types**, **return** and **return types**::
	:parameter: index_path, path of index
	:type: string
        :return: None

    :func:`create_index` - creates an index with defined schema

    - **parameters**, **types**, **return** and **return types**::

        :return: None
       
    :func:`open_index` - opens up the index directory for reading the index

    - **parameters**, **types**, **return** and **return types**::

        :return: returns the pointer to the index

    :func:`add_document` - adds a document to the index

    - **parameters**, **types**, **return** and **return types**::
	:paramter: url, url representing the file 
	:types: string
	:parameter: title, title of the document being added
	:types: string
	:parameter: content, content of the document
	:types: string
	:parameter: modtime, last modified time stamp
	:types: datetime
        :return: none

    :func:`update_document` - update document to the index

    - **parameters**, **types**, **return** and **return types**::
        :paramter: url, url representing the file 
        :types: string
        :parameter: title, title of the document being added
        :types: string
        :parameter: content, content of the document
        :types: string
        :parameter: modtime, last modified time stamp
        :types: datetime
        :return: none

    :func:`delete_document` - delete document from the index based on url

    - **parameters**, **types**, **return** and **return types**::
        :paramter: url, url representing the file 
        :types: string
	:return: none

    :func:`find` - queries the index based on search query

    - **parameters**, **types**, **return** and **return types**::

        :return: none

    :func:`commit` - commit the document to the index

    - **parameters**, **types**, **return** and **return types**::

        :return: none

    :func:`cancel` - cancels the commit of added documents to index

    - **parameters**, **types**, **return** and **return types**::

        :return: none

 
    """

    ana = analysis.StandardAnalyzer()
    schema = whoosh.fields.Schema(
        url=whoosh.fields.ID(unique=True, stored=True),
        title=whoosh.fields.TEXT(stored=True, phrase=False),
        content=whoosh.fields.TEXT(spelling=True, stored=True, phrase=False), 
        modtime=whoosh.fields.ID())

    def __init__(self, index_path):
        self.path = index_path
        if not os.path.exists(index_path):
            os.makedirs(index_path)

    def create_index(self):
        whoosh.index.create_in(self.path, self.schema)
	self.open_index()

    def open_index(self):
        self._index = whoosh.index.open_dir(self.path)
        self._writer = self._index.writer()
	return self._index

    def add_document(self, url, title, content,modtime):
        self._writer.add_document(
            url=unicode(url),
            title=unicode(title),
            content=unicode(content,errors='ignore'),
	    modtime =unicode(modtime),)

    def update_document(self, url, title, content,modtime):
        self._writer.update_document(
            url=unicode(url),
            title=unicode(title),
            content=unicode(content),
	    modtime = unicode(modtime),)

    def delete_document(self, url):
        self._index.delete_by_term('url', unicode(url))

    _queryparser = whoosh.qparser.QueryParser('content', schema=schema)
    def find(self):
        searcher = self._index.searcher()
	return searcher,self._queryparser

    def commit(self):
        self._writer.commit()

    def cancel(self):
        self._writer.cancel()



[docs]class Searched(object):

    """This class provides abstraction over Whoosh Search Engine.
       It provides clear APIs for 
	1. Indexing
	2. Searching
	3. Adv search - morelikethis
	4. Adv search - didyoumean

    :func:`__init__` - accepts the path where index is stored and 
		the path from where the documents need to be indexed.

    - **parameters**, **types**, **return** and **return types**::
        :parameter: indexpath, path of index
        :type: string
        :parameter: dirpath, path of the root of filesystem
        :type: string
        :return: None

    :func:`indexer` - reads the documents from root of file system, indexes them and commits the content.

    - **parameters**, **types**, **return** and **return types**::

        :return: None

    :func:`searcher` - returns the top 10 results of search query.

    - **parameters**, **types**, **return** and **return types**::

        :return: top 10 results

    :func:`morelikethis` - returns morelikethis results for a document.

    - **parameters**, **types**, **return** and **return types**::
	:parameter: document for morelikethis search
	:type: string
        :return: morelikethis search results

    :func:`didyoumean` - returns didyoumean results for a search query.

    - **parameters**, **types**, **return** and **return types**::
        :parameter: results for didyoumean search
        :type: string
        :return: didyoumean search results

    """

    def __init__(self, indexpath, dirpath, q=None):
	self.indexpath = indexpath
	self.dirpath = dirpath 
	self.q = q

    def indexer(self):
        engine = SearchEngine(self.indexpath)
        engine.create_index()
        for root, dirs, files in os.walk(self.dirpath):
                if len(files) is not 0:
                        for f in files:
                                path = os.path.join(root,f)
                                url = "http://localhost:8888/" + path
                                modtime = os.path.getmtime(path)
                                fh = open(path, 'r')
                                contents = fh.read()
                                fh.close()
                                print "Adding file: %s" %f
                                engine.add_document(url, path, contents, modtime)
	print "Commiting index..."
        engine.commit()
	print "Index Commited"

    def searcher(self):
        engine = SearchEngine(self.indexpath)
       	engine.open_index()
	searcher, _queryparser = engine.find()
	results = searcher.search(query.Term('content', self.q), limit=10)
	docnum = searcher.document_number(url='http://localhost:8888/root/style.css')
	return results


    def morelikethis(self, path):
	engine = SearchEngine(self.indexpath)
	engine.open_index()
	searcher, _queryparser = engine.find()
	url = "http://localhost:8888/" + path
	docnum = searcher.document_number(url=url)
	r = searcher.more_like(docnum, 'content')
	return r

    def didyoumean(self, qstring):
        engine = SearchEngine(self.indexpath)
        engine.open_index()
        searcher, _queryparser = engine.find()
	corrector = searcher.corrector("content")
	r = corrector.suggest(qstring, limit=3)
	return r
